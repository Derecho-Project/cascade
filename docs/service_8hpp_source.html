366 template <typename SubgroupType>
367 struct SubgroupNotificationHandler {
368 // key: object_pool_pathname
369 // value: an option for the handler
370 // The handler for "" key is the default handler, which will always be triggered.
371 std::unordered_map<std::string, std::optional<cascade_notification_handler_t>> object_pool_notification_handlers;
372 mutable std::unique_ptr<std::mutex> object_pool_notification_handlers_mutex;
373
374 SubgroupNotificationHandler():
375 object_pool_notification_handlers_mutex(std::make_unique<std::mutex>()) {}
376
377 template <typename T>
378 inline void initialize(derecho::ExternalClientCaller<SubgroupType,T>& subgroup_caller) {
379 dbg_default_trace("SubgroupNotificationHandler(this={:x}) is initialized for SubgroupType:{}",
380 reinterpret_cast<uint64_t>(this),typeid(SubgroupType).name());
381 subgroup_caller.register_notification_handler(
382 [this](const derecho::NotificationMessage& msg){
383 dbg_default_trace("subgroup notification handler is triggered with this={:x}, msg type={}, size={} bytes",
384 reinterpret_cast<uint64_t>(this), msg.message_type, msg.size);
385 (*this)(msg);
386 });
387 }
388
389 inline void operator ()(const derecho::NotificationMessage& msg) {
390 dbg_default_trace("SubgroupNotificationHandler(this={:x}) is triggered with message_type={:x}, size={} bytes",
391 reinterpret_cast<uint64_t>(this),msg.message_type, msg.size);
392 if (msg.message_type != CASCADE_NOTIFICATION_MESSAGE_TYPE) {
393 return;
394 }
395 // mutils::deserialize_and_run<CascadeNotificationMessage>(nullptr, msg.body,
396 mutils::deserialize_and_run(nullptr, msg.body,
397 [this](const CascadeNotificationMessage& cascade_message)->void {
398 dbg_default_trace("Handling cascade_message: {}. size={} bytes",
399 cascade_message.object_pool_pathname,cascade_message.blob.size);
400 std::lock_guard<std::mutex> lck(*object_pool_notification_handlers_mutex);
401 // call default handler
402 if (object_pool_notification_handlers.find("") !=
403 object_pool_notification_handlers.cend()) {
404 if (object_pool_notification_handlers.at("").has_value()) {
405 (*object_pool_notification_handlers.at(""))(cascade_message.blob);
406 }
407 }
408 // call object pool handler
409 if (object_pool_notification_handlers.find(cascade_message.object_pool_pathname) !=
410 object_pool_notification_handlers.cend()) {
411 if (object_pool_notification_handlers.at(cascade_message.object_pool_pathname).has_value()) {
412 (*object_pool_notification_handlers.at(cascade_message.object_pool_pathname))(cascade_message.blob);
413 }
414 }
415 });
416 }
417 };
