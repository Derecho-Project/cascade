52inline std::vector<std::string> str_tokenizer(const std::string& str, bool prefix_only=false, char separator=PATH_SEPARATOR) {
53 std::vector<std::string> components;
54 std::string::size_type pos=0, spos=0;
55 while (pos != std::string::npos) {
56 pos = str.find(separator,pos);
57 if (pos == std::string::npos) {
58 if (!prefix_only && (spos < str.length())) {
59 components.emplace_back(str.substr(spos));
60 }
61 continue;
62 }
63 // skip leading and consecutive '/'s.
64 if (pos != spos) {
65 components.emplace_back(str.substr(spos,(pos-spos)));
66 }
67 pos = pos + 1;
68 spos = pos;
69 } while (pos != std::string::npos);
70 return components;
71}

391/*
392 * For UDLs:
393 * TLT_ACTION_POST The time when action is inserted into an action_queue for off critical data path processing.
394 * The extra info is defined as follows:
395 * struct {
396 * uint8_t trigger; // 0 - for ordered_put; 1 - for trigger_put
397 * uint8_t stateful; // Following DataFlowGRaph::Statefulness enumerate
398 * uint8_t rsv8_0;
399 * uint8_t rsv8_1;
400 * uint32_t rsv32_0;
401 * } info
402 * TLT_ACTION_FIRE The time when action is fired by a worker thread. The extra info is defined as follows:
403 * struct {
404 * uint32_t worker_id; // the id of the worker thread
405 * uint32_t rsv;
406 * }
407 */
408

410 struct {
411 uint8_t is_trigger;
412 uint8_t stateful;
413 uint8_t rsv8_0;
414 uint8_t rsv8_1;
415 uint32_t rsv32_0;
416 } info;
417 uint64_t uint64_val;
418} ActionPostExtraInfo;
