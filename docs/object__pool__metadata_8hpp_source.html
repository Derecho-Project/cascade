283 template<typename KeyType>
284 inline uint32_t key_to_shard_index(const KeyType& key, const KeyType& affinity_set, uint32_t num_shards, bool check_object_locations = true) const {
285 if constexpr (std::is_convertible_v<KeyType,std::string>) {
286 if (check_object_locations) {
287 if (this->object_locations.find(key) != object_locations.end()) {
288 return object_locations.at(key);
289 }
290 }
291 uint32_t shard_index = 0;
292 switch (sharding_policy) {
293 case HASH:
294 if (static_cast<std::string>(affinity_set).length() > 0) {
295 shard_index = std::hash<std::string>{}(affinity_set) % num_shards;
296 } else {
297 shard_index = std::hash<std::string>{}(key) % num_shards;
298 }
299 break;
300 default:
301 throw derecho::derecho_exception(std::string("Unknown sharding_policy:") + std::to_string(sharding_policy));
302 }
303 return shard_index;
304 } else {
305 throw derecho::derecho_exception(std::string{__PRETTY_FUNCTION__} + " failed with invalid Key Type:" + typeid(KeyType).name());
306 }
307 }
