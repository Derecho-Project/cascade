321 template <typename MessageType>
322 std::unique_ptr<DDSSubscriber<MessageType>> subscribe(
323 ServiceClientAPI& capi,
324 DDSMetadataClient& metadata_service,
325 const std::string& topic,
326 const std::unordered_map<std::string,message_handler_t<MessageType>>& handlers) {
327 // apply the big lock
328 std::lock_guard<std::mutex> lck(registry_mutex);
329 // check if topic is registered.
330 if (registry.find(topic)==registry.cend()) {
331 // create a topic entry.
332 auto topic_info = metadata_service.get_topic(topic);
333 registry.emplace(topic,PerTopicRegistry(topic,topic_info.pathname));
334 // register universal per-topic handler, which dispatches messages to subscriber cores.
335 capi.register_notification_handler(
336 [this,my_id=capi.get_my_id()](const Blob& blob)->void{
337 const DDSMessageHeader* ptr = reinterpret_cast<const DDSMessageHeader*>(blob.bytes);
338 std::string topic(ptr->topic_name,ptr->topic_name_length);
339 dbg_default_trace("notification handler is triggered on topic:{}, size={]",topic,blob.size);
340#if !defined(USE_DDS_TIMESTAMP_LOG)
341 // We cannot get the message id here because the notification is not an object but a blob.
342 TimestampLogger::log(TLT_DDS_SUBSCRIBER_RECV,my_id,0);
343#endif
344 for(auto& subscriber_core:registry.at(topic).registry) {
345 if (subscriber_core.second->online) {
346 subscriber_core.second->post(blob);
347 }
348 }
349 },topic_info.pathname);
350 dbg_default_trace("registered a handler for topic:{} on pathname:{}",topic,topic_info.pathname);
351 // subscribe to a cascade server.
352 _topic_control(capi,topic_info,DDSCommand::SUBSCRIBE);
353 }
354 // register the handlers
355 auto subscriber_core = registry.at(topic).template create_subscriber_core<MessageType>(handlers);
356 // create a Subscriber object wrapping the Subscriber Core
357 return std::make_unique<DDSSubscriberImpl<MessageType>>(subscriber_core);
358 }

365 template <typename MessageType>
366 void unsubscribe(
367 ServiceClientAPI& capi,
368 DDSMetadataClient& metadata_service,
369 const DDSSubscriber<MessageType>& subscriber) {
370 // apply the big lock
371 std::lock_guard<std::mutex> lck(registry_mutex);
372 const DDSSubscriberImpl<MessageType>* impl = dynamic_cast<const DDSSubscriberImpl<MessageType>*>(&subscriber);
373 // test the existence of corresponding subscriber core.
374 if (registry.find(impl->core->topic) == registry.cend()) {
375 dbg_default_warn("unsubscribe abort because subscriber's topic '{}' does not exist in registry.", impl->core->topic);
376 return;
377 }
378 if (registry.at(impl->core->topic).registry.find(impl->core->index) == registry.at(impl->core->topic).registry.cend()) {
379 dbg_default_warn("unsubscribe abort because subscriber's index '{}' does not exist in the per topic registry.", impl->core->index);
380 return;
381 }
382 // remove the subscriber core
383 registry.at(impl->core->topic).registry.erase(impl->core->index);
384 if (registry.at(impl->core->topic).registry.empty()) {
385 auto topic_info = metadata_service.get_topic(impl->core->topic);
386 // unsubscribe from a cascade server
387 _topic_control(capi,topic_info,DDSCommand::UNSUBSCRIBE);
388 // remove the topic entry
389 registry.erase(impl->core->topic);
390 }
391 }
