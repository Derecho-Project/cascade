193template<typename SubgroupType>
194bool PerfTestClient::perf_put(PutType put_type,
195 const std::string& object_pool_pathname,
196 ExternalClientToCascadeServerMapping ec2cs,
197 double read_write_ratio,
198 uint64_t ops_threshold,
199 uint64_t duration_secs,
200 const std::string& output_filename) {
201 debug_enter_func_with_args("object_pool_pathname={},ec2cs={},read_write_ratio={},ops_threshold={},duration_secs={},output_filename={}",
202 object_pool_pathname,static_cast<uint32_t>(ec2cs),read_write_ratio,ops_threshold,duration_secs,output_filename);
203 bool ret = true;
204 // 1 - decides on shard membership policy for the "policy" and "user_specified_node_ids" argument for rpc calls.
205 ShardMemberSelectionPolicy policy = ShardMemberSelectionPolicy::Random;
206 auto object_pool = capi.find_object_pool(object_pool_pathname);
207 if (!object_pool.is_valid() || object_pool.is_null()) {
208 throw derecho::derecho_exception("Cannot find object pool:" + object_pool_pathname);
209 }
210 uint32_t number_of_shards = capi.get_number_of_shards<SubgroupType>(object_pool.subgroup_index);
211 std::map<std::pair<std::string,uint16_t>,std::vector<node_id_t>> user_specified_node_ids;
212 for (const auto& kv:connections) {
213 user_specified_node_ids.emplace(kv.first,std::vector<node_id_t>{number_of_shards});
214 }
215 switch(ec2cs) {
216 case ExternalClientToCascadeServerMapping::FIXED:
217 policy = ShardMemberSelectionPolicy::UserSpecified;
218 for(uint32_t shard_index=0;shard_index<number_of_shards;shard_index++) {
219 auto shard_members = capi.template get_shard_members<SubgroupType>(object_pool.subgroup_index,shard_index);
220 uint32_t connection_index = 0;
221 for(const auto& kv:connections) {
222 user_specified_node_ids[kv.first].at(shard_index) = shard_members.at(connection_index%shard_members.size());
223 connection_index ++;
224 }
225 }
226 break;
227 case ExternalClientToCascadeServerMapping::RANDOM:
228 policy = ShardMemberSelectionPolicy::Random;
229 break;
230 case ExternalClientToCascadeServerMapping::ROUNDROBIN:
231 policy = ShardMemberSelectionPolicy::RoundRobin;
232 break;
233 };
234 // 2 - send requests and wait for response
235
236 std::string rpc_cmd{};
237 switch(put_type){
238 case PutType::PUT:
239 rpc_cmd = "perf_put_to_objectpool";
240 break;
241 case PutType::PUT_AND_FORGET:
242 rpc_cmd = "perf_put_and_forget_to_objectpool";
243 break;
244 case PutType::TRIGGER_PUT:
245 rpc_cmd = "perf_trigger_put_to_objectpool";
246 break;
247 }
248 int64_t start_sec = static_cast<int64_t>(get_walltime())/1e9 + 5; // wait for 5 second so that the rpc servers are started.
249
250 std::map<std::pair<std::string,uint16_t>,std::future<RPCLIB_MSGPACK::object_handle>> futures;
251 for (auto& kv: connections) {
252 futures.emplace(kv.first,kv.second->async_call(rpc_cmd,
253 object_pool_pathname,
254 static_cast<uint32_t>(policy),
255 user_specified_node_ids.at(kv.first),
256 read_write_ratio,
257 ops_threshold,
258 start_sec,
259 duration_secs,
260 output_filename));
261 }
262
263 ret = check_rpc_futures(std::move(futures));
264
265 // 3 - flush server timestamps
266 capi.template dump_timestamp(output_filename,object_pool_pathname);
267
268 debug_leave_func();
269 return ret;
270}
