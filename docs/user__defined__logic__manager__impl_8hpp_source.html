144 void load_and_initialize_dlls(CascadeContext<CascadeTypes...>* ctxt) {
145 std::ifstream config(UDL_DLLS_CONFIG);
146 //step 1: test if UDL_DLLS_CONFIG exists or not.
147 if (!config.good()) {
148 dbg_default_warn("{} failed because {} does not exist or is not readable.", __PRETTY_FUNCTION__, UDL_DLLS_CONFIG);
149 return;
150 }
151 //step 2: load .so files one by one.
152 std::string dll_file_path;
153 while(std::getline(config,dll_file_path)) {
154 auto udl = std::make_unique<DLLUserDefinedLogic<CascadeTypes...>>(dll_file_path);
155 if (udl->is_valid()) {
156 udl->initialize(ctxt);
157 dbg_default_trace("Successfully load dll udl:{}",dll_file_path,udl->id);
158 udl_map[udl->id] = std::move(udl);
159 } else {
160 dbg_default_error("Failed loading dll udl:{}.", dll_file_path);
161 }
162 }
